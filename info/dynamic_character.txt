Говорящиий с возможностью тзиенть некоторые его параметры в любое время
```python
init:
    $ loli_name = u"Лоли"
    $ loli = DynamicCharacter("loli_name", what_color=(255,211,155), who_color=(238,44,44))


label my_label:
    #Именя говорящего "Лоли"
    loli 'Привет'

    $ loli_name = u'Маша'
    #Имя говорящего "Маша"
    loli "А тебя как зовут?"
```
